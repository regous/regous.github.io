C#：OpenFileDialog 对话框封装

打开对话框让用户选择文件、保存文件，是常用的功能，用法也不是很简单的，最主要的就是文件过滤器需要手写。因此封装一个类来专门完成这项工作。

/// <summary>
    /// 一些文件相关的操作
    /// </summary>
    public static class Browser
    {
        /// <summary>
        /// 选择文件
        /// </summary>
        /// <param name="Filter">文件扩展名过滤器</param>
        /// <param name="InitialDirectory">文件选择对话框初始目录</param>
        /// <param name="Title">文件选择对话框标题</param>
        /// <returns>
        /// 
        /// </returns>
        public static string PickFile(string Filter, string InitialDirectory, string Title)
        {
            Microsoft.Win32.OpenFileDialog dlg = new Microsoft.Win32.OpenFileDialog();
            if (!string.IsNullOrEmpty(Filter))
                dlg.Filter = Filter;
            if (!string.IsNullOrEmpty(InitialDirectory))
                dlg.InitialDirectory = InitialDirectory;
            if (!string.IsNullOrEmpty(Title))
                dlg.Title = Title;
            dlg.Multiselect = false;
            dlg.CheckFileExists = true;
            if (dlg.ShowDialog() == true)
                return dlg.FileName;
            else
                return null;
        }
        /// <summary>

        public static string PickFile(FileFilter Filter, string InitialDirectory, string Title)
        {
            string filefilter = "";
            if (Filter != null)
                filefilter = Filter.ToString();
            return PickFile(filefilter, InitialDirectory, Title);
        }

        /// <summary>
        /// 选择文件夹
        /// </summary>
        /// <param name="InitialDirectory">文件选择对话框初始目录</param>
        /// <param name="Title">文件选择对话框标题</param>
        /// <returns></returns>
        public static string PickFolder(string InitialDirectory, string Title)
        {
            Microsoft.WindowsAPICodePack.Dialogs.CommonOpenFileDialog dlg = new Microsoft.WindowsAPICodePack.Dialogs.CommonOpenFileDialog();
            dlg.IsFolderPicker = true; //set to false if need to select files
            if (!string.IsNullOrEmpty(Title))
                dlg.Title = Title;
            else
                dlg.Title = "选择文件夹";
            if (!string.IsNullOrEmpty(InitialDirectory))
                dlg.InitialDirectory = InitialDirectory;
            dlg.EnsurePathExists = true;
            dlg.AddToMostRecentlyUsedList = true;
            //d.InitialDirectory = Environment.GetFolderPath(Environment.SpecialFolder.MyDocuments);
            Microsoft.WindowsAPICodePack.Dialogs.CommonFileDialogResult result = dlg.ShowDialog(Wpf.Win32WindowExtension.GetActiveWindow());
            if (result == Microsoft.WindowsAPICodePack.Dialogs.CommonFileDialogResult.Ok)
            {
                return dlg.FileName;
            }
            else
            {
                return null;
            }
        }

/// <summary>
    /// 构建 Filter 
    /// </summary>
    /// <remarks>
    ///
    /// </remarks>
    /// <example>
    /// FileFilter.Create().Add("Excel 工作簿",".xlsx").Add("Excel 2000 文档","xls").ToString() -> "Excel 文档|*.xlsx"|Excel 2000 文档|*.xls
    /// 配合Browser使用
    /// 
    /// Browser.PickFile(FileFilter.Add("Excel 文档",".xlsx").ToString(),null,null);
    /// </example>
    public class FileFilter
    {
        #region 常用的类型
        public static FileFilter AllFiles => FileFilter.Create().Add(FNE.AllFiles);
        public static FileFilter Xlsx => FileFilter.Create().Add(FNE.Xlsx);
        public static FileFilter Xls => FileFilter.Create().Add(FNE.Xls);
        public static FileFilter Docx => FileFilter.Create().Add(FNE.Docx);
        public static FileFilter Doc => FileFilter.Create().Add(FNE.Doc);
        public static FileFilter Csv => FileFilter.Create().Add(FNE.Csv);
        public static FileFilter Txt => FileFilter.Create().Add(FNE.Txt);
        public static FileFilter Text => FileFilter.Create().Add(FNE.Text);
        public static FileFilter TxtOrText => FileFilter.Create().Add(
            FilenameExtension.MergeAll(
                "文本文件",
                FilenameExtension.Txt,
                FilenameExtension.Text)
            );

        public static FileFilter Jpg => FileFilter.Create().Add(FNE.Jpg);
        public static FileFilter Jpeg => FileFilter.Create().Add(FNE.Jpeg);
        public static FileFilter JpgOrJpeg => FileFilter.Create().Add(
            FilenameExtension.MergeAll(
                "Jpg 图片",
                FilenameExtension.Jpg,
                FilenameExtension.Jpeg)
            );
        public static FileFilter Png => FileFilter.Create().Add(FNE.Png);
        public static FileFilter Gif => FileFilter.Create().Add(FNE.Gif);
        public static FileFilter Zip => FileFilter.Create().Add(FNE.Zip);
        public static FileFilter Rar => FileFilter.Create().Add(FNE.Rar);
        public static FileFilter _7z => FileFilter.Create().Add(FNE._7z);
        public static FileFilter Mdb => FileFilter.Create().Add(FNE.Mdb);
        public static FileFilter Accdb => FileFilter.Create().Add(FNE.Accdb);
        public static FileFilter MdbOrAccdb => FileFilter.Create().Add(
            FilenameExtension.MergeAll(
                "Access 数据库",
                FilenameExtension.Mdb,
                FilenameExtension.Accdb)
            );
        public static FileFilter SqliteDb => FileFilter.Create().Add(FNE.SqliteDb);
        public static FileFilter SqliteDb3 => FileFilter.Create().Add(FNE.SqliteDb3);
        public static FileFilter SqliteSqlite => FileFilter.Create().Add(FNE.SqliteSqlite);
        public static FileFilter Sqlite => FileFilter.Create().Add(FilenameExtension.MergeAll("Sqlite 数据库",
            FilenameExtension.SqliteDb, FilenameExtension.SqliteDb3, FilenameExtension.SqliteSqlite));
        #endregion
        public static FileFilter Create() => new FileFilter();
        public static FileFilter Create(string name, string extension) => FileFilter.Create().Add(name, extension);
        public static FileFilter Create(FilenameExtension fne) => FileFilter.Create().Add(fne);
        public static FileFilter Create(MultiFilenameExtension mfne) => FileFilter.Create().Add(mfne);
        public static FileFilter Create(FNE fne) => FileFilter.Create().Add(fne);

        public static FileFilter I => new FileFilter();
        public static implicit operator string(FileFilter ff) => ff.ToString();

        private List<FilenameExtension> Extensions = new List<FilenameExtension>();

        /// <summary>
        /// 添加一种类型
        /// </summary>
        /// <param name="name">类型名，如（jpeg 图片）</param>
        /// <param name="extension">类型扩展名，如(jpg 或 .jpg 或 *.jpg 或 jpeg...)</param>
        /// <returns></returns>
        public FileFilter Add(string name, string extension)
        {
            return Add(new FilenameExtension(name, extension));
        }

        public FileFilter Add(string name, params string[] extensions)
        {
            if (extensions == null || extensions.Length == 0) return this;
            return Add(name, FilenameExtension.CombineWildcardExpression(extensions));
        }

        public FileFilter Add(FilenameExtension ext)
        {
            Extensions.Add(ext);
            return this;
        }

        public FileFilter Add(MultiFilenameExtension ext)
        {
            Extensions.AddRange(ext.FilenameExtensions);
            return this;
        }

        public FileFilter Add(FNE fne)
        {
            FilenameExtension fe = FilenameExtension.GetFilenameExtension(fne);
            if (fe != null)
                this.Add(fe);
            return this;
        }
        public FileFilter Add(FileFilter filter)
        {
            if (filter == null) return this;
            foreach(var ext in filter.Extensions)
            {
                Add(ext);
            }
            return this;
        }
        private bool AllFile = false;
        public FileFilter SetAll()
        {
            AllFile = true;
            return this;
        }
        public FileFilter AddAll()
        {
            FilenameExtension fe = FilenameExtension.GetFilenameExtension(FNE.AllFiles);
            if (fe != null)
                this.Add(fe);
            return this;
        }
        public override string ToString()
        {
            StringBuilder aStringBuilder = new StringBuilder();
            foreach (var item in Extensions)
            {
                aStringBuilder.Append(item.Name);
                aStringBuilder.Append('|');
                if (item.Extension[0] == '*')
                    aStringBuilder.Append(item.Extension);
                else if (item.Extension[0] == '.')
                    aStringBuilder.Append($"*{item.Extension}");
                else
                    aStringBuilder.Append($"*.{item.Extension}");
                aStringBuilder.Append('|');
            }
            if (AllFile || Extensions.Count == 0)
                aStringBuilder.Append("所有文件|*.*|");
            aStringBuilder.Remove(aStringBuilder.Length - 1, 1);
            return aStringBuilder.ToString();
        }

        public static FileFilter operator +(FileFilter left, FileFilter right)
        {
            return FileFilter.Create().Add(left).Add(right);
        }

    }

    /// <summary>

    [DebuggerDisplay("{Name} {Extension}")]
    public class FilenameExtension
    {
        #region 常用的文件类型
        public static FilenameExtension AllFiles => new FilenameExtension("所有文件", ".*");
        public static FilenameExtension Xlsx => new FilenameExtension("Excel 工作簿", ".xlsx");
        public static FilenameExtension Xls => new FilenameExtension("Excel 97-2000 工作簿", ".xls");
        public static FilenameExtension Docx => new FilenameExtension("Word 文档", ".docx");
        public static FilenameExtension Doc => new FilenameExtension("Word 97-2000 文档", ".doc");
        public static FilenameExtension Dat => new FilenameExtension("文本数据文件", ".dat");
        public static FilenameExtension Csv => new FilenameExtension("CSV (逗号分割符)", ".csv");
        public static FilenameExtension Txt => new FilenameExtension("文本文件", ".txt");
        public static FilenameExtension Text => new FilenameExtension("文本文件", ".text");
        public static FilenameExtension Jpg => new FilenameExtension("JPEG 图片", ".jpg");
        public static FilenameExtension Jpeg => new FilenameExtension("JPEG 图片", ".jpeg");
        public static FilenameExtension Png => new FilenameExtension("PNG 图片", ".png");
        public static FilenameExtension Gif => new FilenameExtension("GIF 格式", ".gif");
        public static FilenameExtension Zip => new FilenameExtension("ZIP 压缩文件", ".zip");
        public static FilenameExtension Rar => new FilenameExtension("RAR 压缩文件", ".rar");
        public static FilenameExtension _7z => new FilenameExtension("7z 压缩文件", ".7z");
        public static FilenameExtension Mdb => new FilenameExtension("Access 2000 数据库", ".mdb");
        public static FilenameExtension Accdb => new FilenameExtension("Access 数据库", ".accdb");
        public static FilenameExtension SqliteDb => new FilenameExtension("Sqlite 数据库", ".db");
        public static FilenameExtension SqliteDb3 => new FilenameExtension("Sqlite 数据库", ".db3");
        public static FilenameExtension SqliteSqlite => new FilenameExtension("Sqlite 数据库", ".sqlite");
        public static FilenameExtension GetFilenameExtension(FNE fne)
        {
            switch (fne)
            {
                case FNE.AllFiles: return AllFiles;
                case FNE.Xlsx: return Xlsx;
                case FNE.Xls: return Xls;
                case FNE.Docx: return Docx;
                case FNE.Doc: return Doc;
                case FNE.Csv: return Csv;
                case FNE.Txt: return Txt;
                case FNE.Text: return Text;
                case FNE.Jpg: return Jpg;
                case FNE.Jpeg: return Jpeg;
                case FNE.Png: return Png;
                case FNE.Gif: return Gif;
                case FNE.Zip: return Zip;
                case FNE.Rar: return Rar;
                case FNE._7z: return _7z;
                case FNE.Mdb: return Mdb;
                case FNE.Accdb: return Accdb;
                case FNE.SqliteDb: return SqliteDb;
                case FNE.SqliteDb3: return SqliteDb3;
                case FNE.SqliteSqlite: return SqliteSqlite;
                case FNE.Dat:return Dat;
                default:
                    break;
            }
            return null;
        }
        #endregion
        public FilenameExtension(string name, string extension)
        {
            Name = name;
            Extension = extension;
        }
        /// <summary>
        /// 文件类型名
        /// </summary>
        public string Name { get; private set; }
        /// <summary>
        /// 文件扩展名
        /// </summary>
        public string Extension { get; private set; }
        /// <summary>
        /// 获取通配符表达式
        /// </summary>
        /// <returns>
        /// 形如 *.jpg 或 *.jpg;*.png
        /// </returns>
        public string GetWildcardExpression() => GetWildcardExpression(Extension);

        /// <summary>
        /// 连接两个文件扩展名
        /// </summary>
        /// <param name="left"></param>
        /// <param name="right"></param>
        /// <returns></returns>
        public static MultiFilenameExtension operator +(FilenameExtension left, FilenameExtension right)
        {
            return FilenameExtension.Combine(left, right);
        }

        public static MultiFilenameExtension operator |(FilenameExtension left, FilenameExtension right)
        {
            return FilenameExtension.Combine(left, right);
        }

        public static MultiFilenameExtension Combine(FilenameExtension left, FilenameExtension right)
        {
            MultiFilenameExtension mfe = new MultiFilenameExtension();
            mfe._FilenameExtensions.Add(left);
            mfe._FilenameExtensions.Add(right);
            return mfe;
        }



        public static MultiFilenameExtension Combine(FilenameExtension left, MultiFilenameExtension right)
        {
            MultiFilenameExtension mfe = new MultiFilenameExtension();
            mfe._FilenameExtensions.Add(left);
            mfe._FilenameExtensions.AddRange(right.FilenameExtensions);
            return mfe;
        }

        public static FilenameExtension operator *(FilenameExtension left, FilenameExtension right)
        {
            return FilenameExtension.MergeAll(left.Name, left, right);
        }

        public static FilenameExtension operator &(FilenameExtension left, FilenameExtension right)
        {
            return FilenameExtension.MergeAll(left.Name, left, right);
        }

        public static FilenameExtension operator *(string name, FilenameExtension right)
        {
            return FilenameExtension.MergeAll(name, right);
        }

        public static FilenameExtension operator &(string name, FilenameExtension right)
        {
            return FilenameExtension.MergeAll(name, right);
        }
        /// <summary>
        /// 连接另外一个文件扩展名对象
        /// </summary>
        /// <param name="right">另外文件扩展名对象</param>
        /// <param name="name">连接后显示文件类型名</param>
        /// <returns>连接后的文件扩展对象</returns>
        public FilenameExtension Merge(string name, params FilenameExtension[] fnes)
        {
            List<FilenameExtension> fes = new List<FilenameExtension>() { this };
            fes.AddRange(fnes);
            return MergeAll(name, fnes);
        }

        public static FilenameExtension MergeAll(string name, params FilenameExtension[] fnes)
        {
            if (fnes.Length == 1) return new FilenameExtension(name, fnes[0].Extension);
            StringBuilder builder = new StringBuilder();
            foreach (var fne in fnes)
            {
                if (fne == null) continue;
                builder.Append(fne.GetWildcardExpression());
                builder.Append(';');
            }
            if (builder.Length > 0) builder.Remove(builder.Length - 1, 1);
            else return null;
            if (name == null)
            {
                var dfne = fnes.FirstOrDefault(x => x != null);
                if (dfne != null)
                    name = dfne.Name;
            }
            if (name == null) name = "文件";
            return new FilenameExtension(name, builder.ToString());
        }

        /// <summary>
        /// 获取通配符表达式
        /// </summary>
        /// <returns>
        /// 形如 *.jpg
        /// </returns>
        public static string GetWildcardExpression(string extension)
        {
            if (string.IsNullOrEmpty(extension)) return "*.*";
            if (extension == "*" || extension == ".") return "*.*";
            if (extension.Length == 1) return $"*.{extension}";
            if (extension[0] == '*' && extension[1] == '.') return extension;
            if (extension[0] == '.') return $"*{extension}";
            return $"*.{extension}";
        }

        public static string CombineWildcardExpression(params string[] extensions)
        {
            if (extensions == null || extensions.Length == 0) return "";
            StringBuilder builder = new StringBuilder();
            foreach (string extension in extensions)
            {
                if (string.IsNullOrEmpty(extension)) continue;
                builder.Append(GetWildcardExpression(extension));
                builder.Append(';');
            }
            if (builder.Length != 0)
                builder.Remove(builder.Length - 1, 1);
            return builder.ToString();
        }
    }

        public static MultiFilenameExtension Merge(MultiFilenameExtension left, MultiFilenameExtension right)
        {
            MultiFilenameExtension mfe = new MultiFilenameExtension();
            mfe._FilenameExtensions.AddRange(left._FilenameExtensions);
            mfe._FilenameExtensions.AddRange(right._FilenameExtensions);
            return mfe;
        }
        public static MultiFilenameExtension Merge(MultiFilenameExtension left, FilenameExtension right)
        {
            MultiFilenameExtension mfe = new MultiFilenameExtension();
            mfe._FilenameExtensions.AddRange(left._FilenameExtensions);
            mfe._FilenameExtensions.Add(right);
            return mfe;
        }
