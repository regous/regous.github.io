lines = []


def printl(str):
    print(str)
    lines.append(str)


def read_registry(expression='#HKEY_LOCAL_MACHINE\SOFTWARE\GitForWindows:InstallPath#'):
    match = re.match('#(.+):(.+)#', expression) if ':' in expression else re.match('#(.+)#', expression)
    if match is None:
        return expression
    else:
        try:
            count = len(match.groups())
            key = match.groups()[0]
            vkey = ''
            if count > 1:
                vkey = match.groups()[1]
            parts = key.split('\\')
            hkey = None
            if parts[0] == 'HKEY_LOCAL_MACHINE':
                hkey = winreg.HKEY_LOCAL_MACHINE
            elif parts[0] == 'HKEY_CURRENT_USER':
                hkey = winreg.HKEY_CURRENT_USER
            elif parts[0] == 'HKEY_CLASSES_ROOT':
                hkey = winreg.HKEY_CLASSES_ROOT
            regkey = winreg.OpenKey(hkey, '\\'.join(parts[1:]))
            value, type = winreg.QueryValueEx(regkey, vkey)
            return value
        except Exception as e:
            print(e)
            return ''


def parse(item, expression):
    while True:
        match = re.search(r'%.+?%', expression)
        if match is None:
            break
        value = match.group(0)
        virable = os.environ[value[1:-1]]
        expression = expression.replace(value, virable, 1)

    while True:
        if '#' in expression:
            pass
        match = re.search(r'#.+?#', expression)
        if match is None:
            break
        value = match.group(0)
        content = read_registry(value)
        expression = expression.replace(value, content, 1)

    while True:
        match = re.search(r'\$\w+?\$', expression)
        if match is None:
            return expression
        value = match.group(0)
        if value[1:-1] in item:
            expression = expression.replace(value, parse(item, item[value[1:-1]]), 1)
        else:
            return expression


def escape(expression):
    return expression.replace('\\', '\\\\').replace('"', '\\"')


if __name__ == '__main__':
    flag = True
    if len(sys.argv) > 1:
        if sys.argv[1].lower() == '-a':
            flag = True
        elif sys.argv[1].lower() == '-d':
            flag = False
        else:
            flag = True

    printl('Windows Registry Editor Version 5.00')
    printl('')
    if flag:
        for path in paths:
            printl(r'[HKEY_CLASSES_ROOT\{0}\shell\{1}]'.format(path, com_name))
            printl('"SubCommands"=""')
            printl('"MUIVerb"="{0}"'.format(com_title))
            printl('')
            printl(r'[HKEY_CLASSES_ROOT\{0}\shell\{1}\Shell]'.format(path, com_name))

            for item in items:
                printl('')
                printl(r'[HKEY_CLASSES_ROOT\{0}\shell\{1}\Shell\{2}]'.format(path, com_name, item['name']))
                printl(r'@ = "{0}"'.format(item['title']))
                if len(item['icon']) != 0:
                    icon = parse(item, item['icon'])
                    icon = escape(icon)
                    printl(r'"Icon" = "{0}"'.format(icon))
                printl('')
                printl(r'[HKEY_CLASSES_ROOT\{0}\shell\{1}\Shell\{2}\command]'.format(path, com_name, item['name']))
                key = 'command_background' if 'background' in path.lower() else 'command'
                command = ''
                if key in item:
                    command = item[key]
                else:
                    command = item['command']
                command = parse(item, command)
                command = escape(command)
                printl(r'@ = "{0}"'.format(command))
                printl('')
    else:
        for path in paths:
            printl(r'[-HKEY_CLASSES_ROOT\{0}\shell\{1}]'.format(path, com_name))
            printl('')

    regfile = __file__[:-3]
    if flag:
        regfile = __file__[:-3] + '-a'
    else:
        regfile = __file__[:-3] + '-r'
    regfile = regfile + '.reg'
    with open(regfile, mode='w', encoding='ansi') as reg:
        for line in lines:
            reg.write(line)
            reg.write('\n')
    content = input('是否导入注册表(Y)：')
    print(content)
    if content.upper() == 'Y':
        os.system(f'regedit.exe "{regfile}"')
