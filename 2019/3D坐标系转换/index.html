3D坐标系转换

​ 写二维的平面算法比三维要简单多了。有时候条件合适，例如点在同一个平面上，可以把三维的坐标转换二维的坐标，去掉 Z 值，这样就可以用二维的平面算法去处理程序的逻辑。处理完成之后，再把二维的坐标转换回三维的坐标。其实所谓的转换是转换参考坐标系，我把人脑最直观理解的称为世界坐标系，世界坐标系的 Z 值，就是重力的方向，眼睛直面的是 Y 轴的方向，左右则是 X 轴。而转换参考系，比如把参考的平面定义在一面墙上，垂直于墙面的是 Z 轴，水平方向是 X 轴，重力方向是 Y 轴。如前所言，墙面上的一点，如果按世界坐标系，该点带有 Z 值，如果是墙面参考系，则 Z 值是 0，这样如果所有的点都在墙面上，那我们把参考系切换到墙面上，因为 Z 值是 0，则可以忽略 Z 值。处理完成之后，再把结果坐标转换回世界坐标系即可，这样可以简化算法的设计。

核心代码

/// <summary>
/// 创建3维坐标到2维平面坐标的转换矩阵
/// </summary>
/// <param name="xaxis">平面坐标系 X 轴的单位向量</param>
/// <param name="yaxis">平面坐标系 Y 轴的单位向量</param>
/// <param name="zaxis">平面坐标系 Z 轴的单位向量</param>
/// <param name="origin">平面坐标系坐标原点</param>
/// <remarks>
/// 参数都是世界坐标系
/// </remarks>
/// <returns>
/// 转换矩阵
/// </returns>
public static Matrix4x4 CreateTransformationMatrix(
    Vector3 xaxis, 
    Vector3 yaxis, 
    Vector3 zaxis, 
    Vector3 origin)
{
    xaxis = Vector3.Normalize(xaxis);
    yaxis = Vector3.Normalize(yaxis);
    zaxis = Vector3.Normalize(zaxis);
    double d = Vector3.Dot(origin, xaxis) + Vector3.Dot(origin, yaxis) + Vector3.Dot(origin, zaxis);
    d = zaxis.X * origin.X + zaxis.Y * origin.Y + zaxis.Z * origin.Z;
    Matrix4x4 matrix = new Matrix4x4
        (
        xaxis.X, yaxis.X, zaxis.X, 0,
        xaxis.Y, yaxis.Y, zaxis.Y, 0,
        xaxis.Z, yaxis.Z, zaxis.Z, 0,
        -Vector3.Dot(xaxis, origin), -Vector3.Dot(yaxis, origin), -Vector3.Dot(zaxis, origin), 1
        );
    return matrix;
}

/// <summary>
/// 两个坐标系之间的相互转换矩阵
/// </summary>
/// <param name="fromxaxis">from 坐标系的 X 轴法向量</param>
/// <param name="fromyaxis">from 坐标系的 Y 轴法向量</param>
/// <param name="fromzaxis">from 坐标系的 Z 轴法向量</param>
/// <param name="fromorigin">from 坐标系坐标原点</param>
/// <param name="toxaxis">to 坐标系的 X 轴法向量</param>
/// <param name="toyaxis">to 坐标系的 Y 轴法向量</param>
/// <param name="tozaxis">to 坐标系的 Z 轴法向量</param>
/// <param name="toorigin">to 坐标系坐标原点</param>
/// <returns></returns>
public static Matrix4x4 CreateTransformationMatrix(
    Vector3 fromxaxis,
    Vector3 fromyaxis,
    Vector3 fromzaxis,
    Vector3 fromorigin,
    Vector3 toxaxis,
    Vector3 toyaxis,
    Vector3 tozaxis,
    Vector3 toorigin)
{
    Matrix4x4 tofrom = CreateTransformationMatrix(fromxaxis, fromyaxis, fromzaxis, fromorigin);
    Matrix4x4.Invert(tofrom, out var tofrominvert);
    Matrix4x4 toto = CreateTransformationMatrix(toxaxis, toyaxis, tozaxis, toorigin);
    return Matrix4x4.Multiply(tofrominvert, toto);
}

static void TextCoordinateSystemTransformation()
{
    Vector3 fromxaxis = new Vector3(GetRandomInt32(), GetRandomInt32(), GetRandomInt32());
    Vector3 fromyaxis = new Vector3(GetRandomInt32(), GetRandomInt32(), GetRandomInt32());
    Vector3 fromzaxis = new Vector3(GetRandomInt32(), GetRandomInt32(), GetRandomInt32());
    Vector3 fromorigin = new Vector3(GetRandomInt32(), GetRandomInt32(), GetRandomInt32());
    Vector3 toxaxis = new Vector3(GetRandomInt32(), GetRandomInt32(), GetRandomInt32());
    Vector3 toyaxis = new Vector3(GetRandomInt32(), GetRandomInt32(), GetRandomInt32());
    Vector3 tozaxis = new Vector3(GetRandomInt32(), GetRandomInt32(), GetRandomInt32());
    Vector3 toorigin = new Vector3(GetRandomInt32(), GetRandomInt32(), GetRandomInt32());
    Matrix4x4 frommatrix = CoordinateSystemTransformation.CreateTransformationMatrix(fromxaxis, fromyaxis, fromzaxis, fromorigin);
    Matrix4x4 tomatrix = CoordinateSystemTransformation.CreateTransformationMatrix(toxaxis, toyaxis, tozaxis, toorigin);
    Vector3 point = new Vector3(GetRandomInt32(), GetRandomInt32(), GetRandomInt32());
    Vector3 point_from = Vector3.Transform(point, frommatrix);
    Vector3 point_to = Vector3.Transform(point, tomatrix);
    Matrix4x4 from_to = CoordinateSystemTransformation.CreateTransformationMatrix(fromxaxis, fromyaxis, fromzaxis, fromorigin, toxaxis, toyaxis, tozaxis, toorigin);
    Vector3 result = Vector3.Transform(point_from, from_to);
}

static int GetRandomInt32()
{
    System.Security.Cryptography.RNGCryptoServiceProvider cspx =
       new System.Security.Cryptography.RNGCryptoServiceProvider();
    byte[] byteCsp = new byte[4];
    cspx.GetBytes(byteCsp);
    int value = BitConverter.ToInt32(byteCsp, 0);
    return value % 10000;
}
