使用 python 批量修改 WPF XAML 的 Grid.Row 或 Grid.Column 值

WPF Grid 是使用最广范的控件，有时候布局有很多行，中间需要插入1行，然而麻烦的是其对应的 Grid.Row 属性需要依次修改，改得多了就烦了，于是用 python 做了一个批量修改这种值的脚本。

使用脚本首先要保证电脑安装了 python3 和 pyqt。

main.py

# encoding=utf-8
import sys, cv2, numpy as np
from ui_textinputer import Ui_TextInputer
from PyQt5.QtWidgets import QApplication, QWidget, QFileDialog
from PyQt5.QtGui import QKeyEvent, QHideEvent, QFontDatabase, QTextCursor,QImage, QPixmap
import traceback
import os
import PyQt5.QtCore as QtCore
import re


class WpfXamlProcessor:
    def __init__(self, xaml='', row_delta=0, col_delta=0):
        self.xaml = xaml
        self.row_delta = row_delta
        self.col_delta = col_delta

    def replace(self, s, isrow=True):
        pattern = re.compile('Grid\.Row="(\d+?)"') if isrow else re.compile('Grid\.Column="(\d+?)"')
        pos = 0
        while True:
            match = pattern.search(s, pos)
            if match is None:
                return s
            oldvalue = int(match.group(1))
            newvalue = oldvalue + (self.row_delta if isrow else self.col_delta)
            s = '{0}{1}{2}'.format(s[:match.span(1)[0]], newvalue, s[match.span(1)[1]:])
            pos = match.span(1)[1]
        return s

    def run(self):
        sources = self.xaml.split('\n')
        result=[]
        for ol in sources:
            ol = self.replace(ol, True)
            ol = self.replace(ol, False)
            print(ol)
            result.append(ol)
        return '\n'.join(result)


class WpfGridEditor(QWidget, Ui_TextInputer):
    def __init__(self):
        super().__init__()
        self.setupUi(self)
        self.setupSignalSlots()

    def accept(self):
        pass

    def showEvent(self, e):
        pass

    def resizeEvent(self, e):
        pass

    def setupSignalSlots(self):
        self.buttonBox.accepted.connect(self.accept)

    def accept(self):
        #self.textEdit.textCursor().hasSelection()
        if self.checkBox.isChecked()==False:
            xaml = self.textEdit.toPlainText()
            wxp = WpfXamlProcessor(xaml, self.rowSpinBox.value(), self.colSpinBox.value())
            xaml = wxp.run()
            print(xaml)
            self.textEdit.setPlainText(xaml)
        else:
            cursor = self.textEdit.textCursor()
            start = cursor.selectionStart()
            end = cursor.selectionEnd()
            if cursor.hasSelection()==False:
                return

            xaml = cursor.selectedText()
            wxp = WpfXamlProcessor(xaml, self.rowSpinBox.value(), self.colSpinBox.value())
            xaml = wxp.run()

            cursor.setPosition(start, QTextCursor.KeepAnchor)
            cursor.setPosition(end, QTextCursor.KeepAnchor)
            cursor.clearSelection()
            cursor.setPosition(start, QTextCursor.KeepAnchor)
            cursor.insertText(xaml)
            cursor.setPosition(start)
            cursor.setPosition(start+len(xaml), QTextCursor.KeepAnchor)
            self.textEdit.setTextCursor(cursor)

def main():
    qApp = QApplication(sys.argv)
    demo = WpfGridEditor()
    demo.show()
    qApp.exec_()

if __name__ == '__main__':
    main()

界面如下：

主体部分是1个文本输入控件，在这个控件里粘贴Xaml文档，然后在最下面的一行选择，选择行增量和列增量，点击OK即可。

如果行增量是2，假设 Grid.Row=“13”，则处理后的结果为 Grid.Row=“15”，增量可以为负，列增量同理。
