C#：调用子进程并获取控制台输出

要干什么

调用一个脚本执行任务，并且将控制台内容重定向到本进程中一个类似控制台的窗口上，并等待子进程完成任务。

关键点

因为进程是WPF进程，为了界面上的平滑，不能阻塞主线程。如果阻塞了主线程，控制台就无法实时输出了进程的输出。所以子进程的启动必须在子线程中，然后主线程，启动一个窗口显示子进程的命令行输出。

实现

调用者类

调用者类使用 BackgroundWorker 类在后台启动子进程并重定向标准输出，在子进程运行的时候抓取子进程的输出，并以事件的方式传回给主线程。

using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Diagnostics;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Sharp.Utils
{
    public class OutputChangedEvent : EventArgs
    {
        internal OutputChangedEvent(string content)
        {
            Content = content;
        }
        public string Content { get; private set; }
    }

    public enum WorkerStatus
    {
        NotStart = -100,
        Cancelled = -99,
        Exception = -98,
        Normal = 0,
        ProcessError
    }


    public class Invoker : IDisposable
    {
        private enum OutputStateType { StandardOutput, StandardError }
        private class OutputState
        {
            public OutputState(OutputStateType type, string data)
            {
                Type = type;
                Data = data;
            }
            public OutputStateType Type { get; private set; }
            public string Data { get; private set; }
        }

        public event EventHandler<OutputChangedEvent> StandardOutputChanged;
        public event EventHandler<OutputChangedEvent> StandardErrorChanged;
        public event EventHandler<RunWorkerCompletedEventArgs> RunWorkerCompleted;

        protected System.Threading.EventWaitHandle WaitHandle = new System.Threading.EventWaitHandle(false, System.Threading.EventResetMode.ManualReset);

        private BackgroundWorker _Worker = new BackgroundWorker();
        public WorkerStatus Status { get; private set; } = WorkerStatus.NotStart;
        public int ExitCode { get; private set; } = int.MinValue;
        public bool IsBusy => _Worker.IsBusy;
        public bool IsCompleted { get; private set; } = false;
        public Invoker()
        {
            InitializeBackgroundWorker();
        }

        private void InitializeBackgroundWorker()
        {
            _Worker.WorkerReportsProgress = true;
            _Worker.WorkerSupportsCancellation = true;
            _Worker.DoWork += new DoWorkEventHandler(DoWork);
            _Worker.ProgressChanged += new ProgressChangedEventHandler(OnProgessChanged);
            _Worker.RunWorkerCompleted += new RunWorkerCompletedEventHandler(OnWorkerCompleted);
        }

        
        /// <summary>
        /// 启动子进程，不等待进程执行完毕
        /// </summary>
        /// <param name="executable"></param>
        /// <param name="workingdirectory"></param>
        /// <param name="cmdparas"></param>
        public virtual void StartAsync(string executable, string workingdirectory, params object[] cmdparas)
        {
            _Worker.RunWorkerAsync(new Tuple<string, string, object[]>(executable, workingdirectory, cmdparas));
        }

        /// <summary>
        /// 启动子进程，并等待进程执行完毕
        /// </summary>
        /// <param name="executable">可执行程序或脚本</param>
        /// <param name="workingdirectory">工作目录</param>
        /// <param name="doevent">() => System.Windows.Forms.Application.DoEvents()</param>
        /// <param name="cmdparas">命令行参数</param>
        /// <returns></returns>
        /// <remarks>
        /// 因为在等待的进程执行完毕的同时不能阻塞主线程，因此必须调用 System.Windows.Forms.Application.DoEvents() 更新 UI，因为这是一个库，不想引用
        /// System.Windows.Forms，因为这个方法从外部传入
        /// </remarks>
        public virtual WorkerStatus Start(string executable, string workingdirectory, Action doevent, params object[] cmdparas)
        {
            StartAsync(executable, workingdirectory, cmdparas);
            System.Threading.Thread.Sleep(100);
            while (!WaitHandle.WaitOne(100))
            {
                doevent();
            }
            return Status;
        }

        //{
        //    await Invoker.Delay(milliseconds);
        //    return "Finished";
        //}

        //private static Task Delay(double milliseconds)

        //    System.Timers.Timer timer = new System.Timers.Timer();
        //    timer.Elapsed += (o, e) => tcs.TrySetResult(true);
        //    timer.Interval = milliseconds;
        //    timer.AutoReset = false;
        //    timer.Start();
        //    return tcs.Task;

        protected virtual void DoWork(object sender, DoWorkEventArgs e)
        {
            try
            {
                Tuple<string, string, object[]> tuple = e.Argument as Tuple<string, string, object[]>;
                if (string.IsNullOrEmpty(tuple.Item1) || !System.IO.File.Exists(tuple.Item1))
                    throw new Exception("执行文件路径不存在");
                ProcessStartInfo startinfo = new ProcessStartInfo
                {
                    FileName = tuple.Item1,
                    UseShellExecute = false,
                    ErrorDialog = true,
                    CreateNoWindow = true,
                    RedirectStandardOutput = true,
                    //RedirectStandardInput = true,
                    RedirectStandardError = true

                };
                if (!string.IsNullOrEmpty(tuple.Item2) && System.IO.Directory.Exists(tuple.Item2))
                    startinfo.WorkingDirectory = tuple.Item2;
                else
                    startinfo.WorkingDirectory = System.IO.Path.GetDirectoryName(tuple.Item1);
                if (tuple.Item3 != null && tuple.Item3.Length > 0)
                {
                    StringBuilder builder = new StringBuilder();
                    foreach (object para in tuple.Item3)
                    {
                        if (para is string ps)
                        {
                            if (ps.Contains(' '))
                            {
                                builder.Append('"');
                                builder.Append(ps);
                                builder.Append('"');
                            }
                            else
                            {
                                builder.Append(ps);
                            }
                        }
                        else
                        {
                            builder.Append(para);
                        }
                        builder.Append(' ');
                    }
                    startinfo.Arguments = builder.ToString();
                }
                Process process = new Process();
                process.StartInfo = startinfo;
                process.OutputDataReceived += (s, args) =>
                {
                    string ss = args.Data;
                    _Worker.ReportProgress(0, new OutputState(OutputStateType.StandardOutput, ss));
                };
                process.ErrorDataReceived += (s, args) =>
                {
                    string ss = args.Data;
                    _Worker.ReportProgress(0, new OutputState(OutputStateType.StandardError, ss));
                };
                process.Start();
                process.BeginOutputReadLine();
                process.BeginErrorReadLine();
                process.WaitForExit();
                int exitcode = process.ExitCode;
                process.Close();
                e.Result = exitcode;
            }
            catch (Exception ex)
            {
                throw ex;
            }
            finally
            {

            }
        }

        protected void OnProgessChanged(object sender, ProgressChangedEventArgs e)
        {
            if (e.UserState is OutputState state)
            {
                if (state.Type == OutputStateType.StandardOutput)
                    StandardOutputChanged?.Invoke(this, new OutputChangedEvent(state.Data));
                else if (state.Type == OutputStateType.StandardError)
                    StandardErrorChanged?.Invoke(this, new OutputChangedEvent(state.Data));
            }
        }

        protected void OnWorkerCompleted(object sender, RunWorkerCompletedEventArgs e)
        {
            if (e.Cancelled)
            {
                Status = WorkerStatus.Cancelled;
            }
            if (e.Error != null)
            {
                Status = WorkerStatus.Exception;
            }
            Status = WorkerStatus.Normal;
            if (e.Result is int exitcode)
                ExitCode = exitcode;
            WaitHandle.Set();
            RunWorkerCompleted?.Invoke(this, e);
            IsCompleted = true;
        }
        public void Stop()
        {
            _Worker.CancelAsync();
        }

        public void Dispose()
        {
            _Worker.Dispose();
            _Worker = null;
        }
    }
}

调用者主要有两个公共方法：

  • StartAsync
  • Start

StartAsync 是调用子进程后就返回，不等待子进程执行完毕，Start 则是等待子进程执行完毕后再返回。

调用者有两个事件：

  • public event EventHandler StandardOutputChanged;
  • public event EventHandler StandardErrorChanged;

监听这两个事件，即可获取子进程的输出。

using System.Collections.ObjectModel;
using System.Windows;

namespace Sharp.Utils.Wpf.Terminal.Demo
{
    /// <summary>
    /// MainWindow.xaml 的交互逻辑
    /// </summary>
    public partial class MainWindow : Window
    {
        public MainWindow()
        {
            InitializeComponent();
            DataContext = this;
            this.Loaded += MainWindow_Loaded;
        }

        private void MainWindow_Loaded(object sender, RoutedEventArgs e)
        {
            Entries = new ObservableCollection<string>();
        }

        private void Invoker_StandardOutputChanged(object sender, OutputChangedEvent e)
        {
            Entries.Add(e.Content);            
        }

        public ObservableCollection<string> Entries
        {
            get { return (ObservableCollection<string>)GetValue(EntriesProperty); }
            set { SetValue(EntriesProperty, value); }
        }

        // Using a DependencyProperty as the backing store for Entries.  This enables animation, styling, binding, etc...
        public static readonly DependencyProperty EntriesProperty =
            DependencyProperty.Register("Entries", typeof(ObservableCollection<string>), typeof(MainWindow), new PropertyMetadata(null));

        private void Async_Click(object sender, RoutedEventArgs e)
        {
            Invoker invoker = new Invoker();
            invoker.StandardOutputChanged += Invoker_StandardOutputChanged;
            invoker.StartAsync(@"C:\Windows\System32\PING.EXE", "", "127.0.0.1");
           
        }
        private void Sync_Click(object sender, RoutedEventArgs e)
        {
            Invoker invoker = new Invoker();
            invoker.StandardOutputChanged += Invoker_StandardOutputChanged;
            invoker.Start(@"C:\Windows\System32\PING.EXE", "", () => System.Windows.Forms.Application.DoEvents(), "127.0.0.1");
        }
    }
}
