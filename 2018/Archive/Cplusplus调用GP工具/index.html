using namespace std;

IVariantArrayPtr 处理全部要素(IFeatureClassPtr ipFeatureClass)
{
    IVariantArrayPtr ipVariantArray(CLSID_VarArray);
    _variant_t va = cppgp::CreateFeaturesParameter(ipFeatureClass, NULL, NULL);
    ipVariantArray->Add(va);
    ipVariantArray->Add(_variant_t(CComBSTR(_T("..\\Demo\\LINE_全部.shp"))));
    return ipVariantArray;
}

IVariantArrayPtr 处理OID为2的要素(IFeatureClassPtr ipFeatureClass)
{
    IVariantArrayPtr ipVariantArray(CLSID_VarArray);
    IQueryFilterPtr ipQueryFilter(CLSID_QueryFilter);
    ipQueryFilter->put_WhereClause(_bstr_t(_T("OBJECTID = 2 ")));
    _variant_t va = cppgp::CreateFeaturesParameter(ipFeatureClass, ipQueryFilter, NULL);
    ipVariantArray->Add(va);
    ipVariantArray->Add(_variant_t(CComBSTR(_T("..\\Demo\\LINE_OID2.shp"))));
    return ipVariantArray;
}

IVariantArrayPtr 处理OID为1和2的要素(IFeatureClassPtr ipFeatureClass)
{
    IVariantArrayPtr ipVariantArray(CLSID_VarArray);
    std::vector<long> vecOids;
    vecOids.push_back(1);
    vecOids.push_back(2);
    _variant_t va = cppgp::CreateFeaturesParameter(ipFeatureClass, NULL, &vecOids);
    ipVariantArray->Add(va);
    ipVariantArray->Add(_variant_t(CComBSTR(_T("..\\Demo\\LINE_OID12.shp"))));
    return ipVariantArray;
}

BOOL cppgp::GetSelectionLayer(IFeatureClassPtr ipFeatureClass, IQueryFilterPtr ipQueryFilter, IFeatureLayerPtr &ipSelectedFeatureLayer)
{
    HRESULT hr = S_FALSE;
    if (NULL == ipFeatureClass)
    {
        return FALSE;
    }
    esriFeatureType eFeatureType;
    hr = ipFeatureClass->get_FeatureType(&eFeatureType);
    if (FAILED(hr))
        return FALSE;
    IFeatureLayerPtr ipFeatureLayer;
    if (esriFTAnnotation == eFeatureType)
    {
        hr = ipFeatureLayer.CreateInstance(CLSID_FDOGraphicsLayer);
    }
    else if (esriFTSimple == eFeatureType
        || esriFTSimpleEdge == eFeatureType
        || esriFTComplexJunction == eFeatureType
        || esriFTSimpleJunction == eFeatureType)
    {
        hr = ipFeatureLayer.CreateInstance(CLSID_FeatureLayer);
    }
    else
        return FALSE;

    if (FAILED(hr) || NULL == ipFeatureLayer)
    {
        return FALSE;
    }
    hr = ipFeatureLayer->putref_FeatureClass(ipFeatureClass);
    if (FAILED(hr))
    {
        return FALSE;
    }
    if (NULL == ipQueryFilter)
    {
        ipSelectedFeatureLayer = ipFeatureLayer;
        return TRUE;
    }

    IFeatureSelectionPtr ipFeatureSelection = (IFeatureSelectionPtr)ipFeatureLayer;
    if (NULL == ipFeatureSelection)
    {
        return FALSE;
    }


    hr = ipFeatureSelection->SelectFeatures(ipQueryFilter, esriSelectionResultNew, VARIANT_FALSE);
    if (FAILED(hr))
    {
        return FALSE;
    }
    IFeatureLayerDefinitionPtr ipFeatureLayerDefinition = (IFeatureLayerDefinitionPtr)ipFeatureLayer;
    if (NULL == ipFeatureLayerDefinition)
    {
        return FALSE;
    }
    BSTR bstrLayerName = NULL;
    hr = ipFeatureLayer->get_Name(&bstrLayerName);
    hr = ipFeatureLayerDefinition->CreateSelectionLayer(bstrLayerName, VARIANT_TRUE, NULL, NULL, &ipSelectedFeatureLayer);
    if (FAILED(hr) || NULL == ipSelectedFeatureLayer)
    {
        return FALSE;
    }

    return TRUE;
}

BOOL cppgp::GetSelectionLayer(IFeatureClassPtr ipFeatureClass, const std::vector<long>& vecOids, IFeatureLayerPtr &ipSelectedFeatureLayer)
{
    HRESULT hr = S_FALSE;
    if (NULL == ipFeatureClass)
    {
        return FALSE;
    }
    esriFeatureType eFeatureType;
    hr = ipFeatureClass->get_FeatureType(&eFeatureType);
    if (FAILED(hr))
        return FALSE;
    IFeatureLayerPtr ipFeatureLayer;
    if (esriFTAnnotation == eFeatureType)
    {
        hr = ipFeatureLayer.CreateInstance(CLSID_FDOGraphicsLayer);
    }
    else if (esriFTSimple == eFeatureType
        || esriFTSimpleEdge == eFeatureType
        || esriFTComplexJunction == eFeatureType
        || esriFTSimpleJunction == eFeatureType)
    {
        hr = ipFeatureLayer.CreateInstance(CLSID_FeatureLayer);
    }
    else
        return FALSE;

    if (FAILED(hr) || NULL == ipFeatureLayer)
    {
        return FALSE;
    }
    hr = ipFeatureLayer->putref_FeatureClass(ipFeatureClass);
    if (FAILED(hr))
    {
        return FALSE;
    }

    IFeatureSelectionPtr ipFeatureSelection = (IFeatureSelectionPtr)ipFeatureLayer;
    if (NULL == ipFeatureSelection)
    {
        return FALSE;
    }
    ISelectionSetPtr ipSelectSet;

    ipFeatureClass->Select(NULL, esriSelectionType::esriSelectionTypeIDSet, esriSelectionOption::esriSelectionOptionEmpty, NULL, &ipSelectSet);
    for (size_t i = 0; i < vecOids.size(); i++)
    {
        ipSelectSet->Add(vecOids[i]);
    }

    ipFeatureSelection->putref_SelectionSet(ipSelectSet);
    IFeatureLayerDefinitionPtr ipFeatureLayerDefinition = (IFeatureLayerDefinitionPtr)ipFeatureLayer;
    if (NULL == ipFeatureLayerDefinition)
    {
        return FALSE;
    }
    BSTR bstrLayerName = NULL;
    hr = ipFeatureLayer->get_Name(&bstrLayerName);
    hr = ipFeatureLayerDefinition->CreateSelectionLayer(bstrLayerName, VARIANT_TRUE, NULL, NULL, &ipSelectedFeatureLayer);
    if (FAILED(hr) || NULL == ipSelectedFeatureLayer)
    {
        return FALSE;
    }

    return TRUE;
}
_variant_t cppgp::CreateFeaturesParameter(IFeatureClassPtr ipFeatureClass, IQueryFilterPtr ipQueryFilter, const std::vector<long>* pvecSelectedOid)
{
    _variant_t varParam;
    ::VariantInit(&varParam);
    if (ipFeatureClass == NULL)
    {
        return varParam;
    }
    if (ipQueryFilter == NULL&&pvecSelectedOid == NULL)
    {
        varParam.punkVal = ipFeatureClass;
        varParam.vt = VT_UNKNOWN;
        varParam.punkVal->AddRef();
        return varParam;
    }
    IFeatureLayerPtr ipSelectedFeatureLayer;
    if (ipQueryFilter != NULL)
    {
        GetSelectionLayer(ipFeatureClass, ipQueryFilter, ipSelectedFeatureLayer);
    }
    else
    {
        GetSelectionLayer(ipFeatureClass, *pvecSelectedOid, ipSelectedFeatureLayer);
    }
    CString strLayer = iPath::GetTempFileName(iPath::GetTempPath(),_T("lyr"));
    ILayerFilePtr ipLayerFile(CLSID_LayerFile);
    ipLayerFile->New(_bstr_t(strLayer));
    ipLayerFile->ReplaceContents(ipSelectedFeatureLayer);
    ipLayerFile->Save();
    varParam = CComBSTR(strLayer);
    return varParam;
}
