Arcpy根据模板创建工作空间

写其他功能的时候，常常修修补补，后来发现，居然改出了一个不错的类，挺通用的，所以记录下来。

Do not talk, show me the code!

import arcpy
import os
import shutil
import zipfile


class Template(object):
    Unknown = 0
    MDB = 1
    GDB = 2
    SHP = 4
    ZIP = 512

    DEFAULT_TYPE = SHP

    def __init__(self, path='', default='shp'):
        '''
        Construct Function
        :param path: the template's path. class will generate workspace by copy template.
        :param default: default workspace type. if the path parameter is empty,this parameter will decide which type of
                            workspace that will be generated.
        '''
        self.path = path
        self._db_type = self._parse_db_type(default)

    def _parse_db_type(self, default):
        if not self.path:
            if isinstance(default, basestring):
                default = default.lower()
                if default == 'mdb':
                    return Template.MDB
                elif default == 'gdb':
                    return Template.GDB
                elif default == 'shp':
                    return Template.SHP
                else:
                    return Template.DEFAULT_TYPE
            return Template.DEFAULT_TYPE
        else:
            if not os.path.exists(self.path):
                raise Exception('template not exists.')
            ext = self.path.split('.')[-1].lower()
            flag = Template.Unknown
            if ext == 'zip':
                flag = Template.ZIP
                fz = zipfile.ZipFile(self.path, 'r')
                fname = fz.namelist()[0]
                ext = fname.split('.')[-1].lower()[:3]

            if ext == 'mdb':
                return flag | Template.MDB
            elif ext == 'gdb':
                return flag | Template.GDB
            else:
                return flag | Template.SHP

    def clone(self, output, name):
        res = ''
        if not self.path:
            res = self.create(output, name)
        else:
            if not self._db_type & Template.ZIP:
                if self._db_type == Template.MDB:
                    res = os.path.join(output, u'{0}.{1}'.format(name, 'mdb'))
                elif self._db_type == Template.GDB:
                    res = os.path.join(output, u'{0}.{1}'.format(name, 'gdb'))
                elif self._db_type == Template.SHP:
                    res = os.path.join(output, name)
                self._copy(self.path, res)
            else:
                on = self._unzip(output)
                res = os.path.join(output, name)
                if self._db_type & Template.GDB:
                    res = res + '.gdb'
                elif self._db_type & Template.MDB:
                    res = res + '.mdb'
                os.rename(on, res)
        return self._decorate(res)

    def create(self, output, name):
        res = os.path.join(output, name)
        if self._db_type & Template.MDB:
            arcpy.CreatePersonalGDB_management(output, name)
        elif self._db_type & Template.GDB:
            arcpy.CreateFileGDB_management(output, name)
        else:
            if not os.path.isdir(res):
                os.mkdir(res)
        return self._decorate(res)

    def extention(self):
        actual_type = self._db_type ^ Template.ZIP if self._db_type & Template.ZIP else self._db_type
        if actual_type == Template.MDB:
            return 'mdb'
        elif actual_type == Template.GDB:
            return 'gdb'
        else:
            return 'shp'

    def _unzip(self, output):
        fz = zipfile.ZipFile(self.path, 'r')
        files = fz.namelist()
        first = files[0]
        for f in files:
            fz.extract(f, output)
        return os.path.join(output, first)

    def _copy(self, src, dst):
        if os.path.isfile(src):
            shutil.copy(src, dst)
        else:
            shutil.copytree(src, dst)

    def _decorate(self, name):
        if self._db_type & Template.SHP:
            return name
        else:
            ext = '.' + self.extention()
            return name if name[-4:].lower() == ext else name + ext

使用方法

Template的构造参数有两个，都是可选的

t = Template(r'E:\GamingSpace\FunAddin\Demo\FunData.mdb')
t.clone(r'E:\GamingSpace\FunAddin\Demo','FunDataClone')
t.create(r'E:\GamingSpace\FunAddin\Demo','FunDataClone_1')
