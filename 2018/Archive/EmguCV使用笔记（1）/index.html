添加 EmguCV 引用

在工具箱里里拖一个 PictureBox 控件到窗口上

修改 PictureBox 的对象类型：

由 System.Windows.Forms.PictureBox 改为 Emgu.CV.UI.ImageBox

哦什么能这样改呢？

因为 Emgu.CV.UI.ImageBox 是从 System.Windows.Forms.PictureBox 派生的。

修改窗口的代码如下：

    public partial class EmguDemo : Form
    {
        private Emgu.CV.VideoCapture _Capture;
        private Emgu.CV.Mat _Frame;
        public EmguDemo()
        {
            InitializeComponent();
            pictureBox.FunctionalMode = Emgu.CV.UI.ImageBox.FunctionalModeOption.Minimum;
        }

        private void EmguDemo_Load(object sender, EventArgs e)
        {
            _Capture = new Emgu.CV.VideoCapture(@"E:\GamingSpace\OpenCVDemo\data\设备-左.mp4");
            _Capture.ImageGrabbed += _Capture_ImageGrabbed;
            _Frame = new Emgu.CV.Mat();
            _Capture.Start();
        }

        void _Capture_ImageGrabbed(object sender, EventArgs e)
        {
            _Capture.Retrieve(_Frame);
            pictureBox.Image = _Frame;
            System.Threading.Thread.Sleep(50);
        }
    }

VideoCapture 是 EmguCV 的视频捕获类，它有多个构造函数，如果调用默认的构造，那么它将捕获电脑摄像头的影像，这时传入一个 .mp4 文件的路径。VideoCapture 的 ImageGrabbed 事件在它获取一个帧时会触发，在响应函数内，通过 VideoCapture 的 Retrieve 方法获取当前帧，并装当前帧绘制到 PictureBox（ImageBox） 上。这里我是直接调用 VideoCapture 的 Start 方法，让它自己去获取图像，如果想自己去控制节奏的话可以调用 QueryFrame 方法，值得注意的是，调用 QueryFrame 方法时同样触发 ImageGrabbed 事件。

本实例下载链接: https://pan.baidu.com/s/1KUh8L06U2-Z1ID23PAlelg 提取码: qyj6

关于 EmguCV 的一些使用经验

OpenCV 的一些常方法由 CvInvoke 引入，大多数的功能都可以通过 Image<TColor,TDepth> 类的成员方法来实现。在OpenCV 中，一个方法的操作可以使用 IplImage 和 CvMat 混用，但在 EmguCV 中这样做经常会导致抛出异常，更好的方法使用 Image<TColor,TDepth> 对象来操作。有必要的时候再转换为矩阵。使用 Image<TColor,TDepth> 的成员方法进行图像分析时，经常会生一些临时变量，重复的分配内存会造成系统效率的低下，这时如果图像的类型兼容，可以 new 一个中间变量，然后使用 CvInvoke 的方法来操作，实际上，Image<TColor,TDepth> 的成员函数经常是 new 一些临时对象出来，如果参数兼容，就是调用 CvInvoke 的方法来进行转换分析。所以如果我们可以确认参数没有问题，那就直接 调用 CvInvoke 的方法来进行操作。这也是与C++环境下很不一样的地方，像我一开始持有的心态是：既然在 C++ 下的代码是可以跑的，那么一模一样翻译成 C# 应该也没有问题吧！但是实际上我是吃了很多的苦头才知道可以这样使用的。

下面的这个类就是是做警戒线分析使用的代码：

using Emgu.CV;
using Emgu.CV.Structure;
using System;
using System.Collections.Generic;
using System.Drawing;
using System.Linq;
using System.Runtime.InteropServices;
using System.Text;
using System.Threading.Tasks;

namespace MotionDemo.Detector
{
    struct Line
    {
        public Point start;
        public Point end;
        public Line(Point s,Point e)
        {
            start = s;
            end = e;
        }
    }
    class GuardLine : IDetect
    {
        Image<Gray, Byte> pFrImg; //提取的前景图像，即运动目标  
        Image<Gray, Byte> pBkImg; //背景图像 
        Image<Gray, Byte> pAbsDiff; //
        Image<Gray, Byte> pThreshold;
        
        public Line RefLine { get; set; }
        MCvScalar s;
        public Mat Detect(Mat pFrame)
        {
            Image<Bgr, Byte> img = OImage<Bgr, Byte>.FromMat(pFrame);

            //Image<Bgr,Byte> img = pFrame.ToImage<Bgr, Byte>(false);
            
            if (pBkImg == null)
            {
                pBkImg = new Image<Gray, Byte>(new Size(pFrame.Width, pFrame.Height));
                pFrImg = new Image<Gray, Byte>(new Size(pFrame.Width, pFrame.Height));
                pAbsDiff = new Image<Gray, Byte>(new Size(pFrame.Width, pFrame.Height));
                pThreshold = new Image<Gray, Byte>(new Size(pFrame.Width, pFrame.Height));
                pBkImg.ConvertFrom<Bgr, Byte>(img);
            }
            else
            {
                //警戒线入侵分析得到新的帧
                var res = CalLine(pFrame,img);
                return res;

            }
            return pFrame;
        }

        private Mat CalLine(Mat pFrame,Image<Bgr,Byte> img)
        {

            MCvScalar s;
            int n = 0;
            int minx = 0;
            int maxx = 0;
            Line line2;
            {
                pFrImg.ConvertFrom<Bgr, Byte>(img);
                //Image<Gray, Byte> pFrMat = pFrImg.AbsDiff(pBkImg);
                //Image<Gray, Byte> pBiImg = pFrMat.ThresholdBinary(new Gray(10), new Gray(255.0));

                CvInvoke.AbsDiff(pFrImg, pBkImg, pAbsDiff);
                CvInvoke.Threshold(pAbsDiff, pThreshold, 10, 255.0, Emgu.CV.CvEnum.ThresholdType.Binary);

                pFrImg.CopyTo(pBkImg);
                minx = Math.Min(RefLine.start.X, RefLine.end.X);
                maxx = Math.Max(RefLine.start.X, RefLine.end.X);
                for (int i = minx; i < maxx; i++)
                {
                    line2.start.X = i;
                    line2.start.Y = 0;
                    line2.end.X = i;
                    line2.end.Y = 9999;
                    Point cross = CrossPoint(RefLine, line2);

                    s = CvInvoke.cvGet2D(pThreshold, cross.Y, cross.X);
                    if (s.V0 == 0xff)//0xff为白色的
                    {
                        n++;
                        if (n > 10) break;
                    }
                }
            }

            //没有目标入侵
            if (n <= 10)
            {
                CvInvoke.Line(pFrame, RefLine.start, RefLine.end, new MCvScalar(0, 255, 0, 0), 1, Emgu.CV.CvEnum.LineType.AntiAlias, 0);
            }
            else //有目标入侵
            {
                CvInvoke.Line(pFrame, RefLine.start, RefLine.end, new MCvScalar(0x00, 0x00, 0xff), 1, Emgu.CV.CvEnum.LineType.AntiAlias, 0);
                CvInvoke.PutText(pFrame, "warning!", new Point(0, 20), Emgu.CV.CvEnum.FontFace.HersheyTriplex, 1, new MCvScalar(255, 0, 0));
            }
            return pFrame;
        }

        private Point CrossPoint(Line line1, Line line2)
        {
            //    if(!SegmentIntersect(line1.start, line1.end, line2.start, line2.end))  
            //    {// segments not cross     
            //        return 0;  
            //    }  
            Point pt = new Point();
            // line1's cpmponent  
            double X1 = line1.end.X - line1.start.X;//b1  
            double Y1 = line1.end.Y - line1.start.Y;//a1  
            // line2's cpmponent  
            double X2 = line2.end.X - line2.start.X;//b2  
            double Y2 = line2.end.Y - line2.start.Y;//a2  
            // distance of 1,2  
            double X21 = line2.start.X - line1.start.X;
            double Y21 = line2.start.Y - line1.start.Y;
            // determinant  
            double D = Y1 * X2 - Y2 * X1;// a1b2-a2b1  
            //   
            if (D == 0)
                return new Point(0, 0);
            // cross point  
            pt.X = (int)((X1 * X2 * Y21 + Y1 * X2 * line1.start.X - Y2 * X1 * line2.start.X) / D);
            // on screen y is down increased !   
            pt.Y = (int)(-(Y1 * Y2 * X21 + X1 * Y2 * line1.start.Y - X2 * Y1 * line2.start.Y) / D);
            // segments intersect.
            if ((Math.Abs(pt.X - line1.start.X - X1 / 2) <= Math.Abs(X1 / 2)) &&
                (Math.Abs(pt.Y - line1.start.Y - Y1 / 2) <= Math.Abs(Y1 / 2)) &&
                (Math.Abs(pt.X - line2.start.X - X2 / 2) <= Math.Abs(X2 / 2)) &&
                (Math.Abs(pt.Y - line2.start.Y - Y2 / 2) <= Math.Abs(Y2 / 2)))
            {
                return pt;
            }
            return new Point(0, 0);
        }
    }
}
