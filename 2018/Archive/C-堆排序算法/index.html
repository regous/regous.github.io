C++堆排序算法

先上代码：

#include "stdafx.h"
#include <iostream>
#include <vector>
#include <functional>
using namespace std;

void printv(const std::vector<int> & vec)
{
    for (auto iter = vec.begin(); iter != vec.end(); iter++)
    {
        cout << *iter << endl;
    }
}

void swap(int& left, int& right)
{
    left ^= right;
    right ^= left;
    left ^= right;
}

std::vector<int> 堆排序(std::vector<int> vec)
{
    std::function<void(std::vector<int>&, int, int)> adjust_heap;
    adjust_heap = [&adjust_heap](std::vector<int>& v, int index, int length)
    {
        int largest = index;
        int temp = (index << 1) + 1;
        if (temp<length&&v[temp]>v[largest])
            largest = temp;
        for (int i = 0; i < 2; i++)
        {
            temp += i;
            if (temp<length&&v[temp]>v[largest])
                largest = temp;
        }
        if (largest != index)
        {
            swap(v[index], v[largest]);
            adjust_heap(v, largest, length);
        }
    };

    std::function<void(std::vector<int>&)> build_heap = [&adjust_heap](std::vector<int>& v)
    {
        for (int i = (v.size() >> 1) - 1 ; i >= 0; i--)
        {
            adjust_heap(v, i, v.size());
        }
    };

    
    build_heap(vec);
    for (int i = vec.size() - 1; i > 0; i--)
    {
        swap(vec[0], vec[i]);
        adjust_heap(vec, 0, i);
    }
    return vec;
}
