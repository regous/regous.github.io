解释：

​ esriICommand从哪来？AO的原命令接口为ICommand，但是因为ICommand与VS SDK定义的接口重名，因此改为esriICommand

#import <esriSystemUI.olb> raw_interfaces_only, raw_native_types, no_namespace, named_guids exclude("ICursor", "IProgressDialog") rename("ICommand", "esriICommand")

USING_MFC宏定义

#define USING_MFC AFX_MANAGE_STATE(AfxGetStaticModuleState());

这是为什么调用这个语句呢？具体与资源切换有关，无论如何，如果要弹出MFC的对话框，必须要进行资源切换，否则对话框创建就会出错。

编译完成后，VS会自动调用regsvr32.exe注册这个库，但是，为了更好的Arcmap支持，需要使用 esriRegasm.exe 注册，如果仅仅是使用regsvr32.exe注册，需要在在Customize里，使用Add From File，加载DLL后才能看到命令，而使用esriRegasm.exe注册，则可以直接看到命令。

除此之外，使用C++编写的IDockableWindowDef对象必须要用esriRegasm.exe注册才能使用，同理可以想到其他更复杂的接口也必须要用esriRegasm.exe才能更好的才Arcmap支持。

#pragma once
#include "resource.h"       // 主符号



#include "DemoCplusplusArcmapAddin_i.h"



#if defined(_WIN32_WCE) && !defined(_CE_DCOM) && !defined(_CE_ALLOW_SINGLE_THREADED_OBJECTS_IN_MTA)
#error "Windows CE 平台(如不提供完全 DCOM 支持的 Windows Mobile 平台)上无法正确支持单线程 COM 对象。定义 _CE_ALLOW_SINGLE_THREADED_OBJECTS_IN_MTA 可强制 ATL 支持创建单线程 COM 对象实现并允许使用其单线程 COM 对象实现。rgs 文件中的线程模型已被设置为“Free”，原因是该模型是非 DCOM Windows CE 平台支持的唯一线程模型。"
#endif

using namespace ATL;


// CDemoToolbar

class ATL_NO_VTABLE CDemoToolbar :
    public CComObjectRootEx,
    public CComCoClass,
    public IDemoToolbar,
    public IToolBarDef
{
public:
    CDemoToolbar()
    {
    }

    DECLARE_REGISTRY_RESOURCEID(IDR_DEMOTOOLBAR)


    BEGIN_COM_MAP(CDemoToolbar)
        COM_INTERFACE_ENTRY(IDemoToolbar)
        COM_INTERFACE_ENTRY(IToolBarDef)
    END_COM_MAP()

    BEGIN_CATEGORY_MAP(__uuidof(CATID_ControlsToolbars))
        IMPLEMENTED_CATEGORY(__uuidof(CATID_MxCommandBars))
    END_CATEGORY_MAP()

    DECLARE_PROTECT_FINAL_CONSTRUCT()

    HRESULT FinalConstruct()
    {
        return S_OK;
    }

    void FinalRelease()
    {
    }

public:
    // IToolBarDef Methods
public:
    STDMETHOD(get_ItemCount)(long * numItems)
    {
        *numItems = 2;
        return S_OK;
    }
    STDMETHOD(GetItemInfo)(long pos, IItemDef * itemDef)
    {
        if (itemDef == NULL)
            return E_POINTER;
        HRESULT hr = S_FALSE;
        switch (pos)
        {
        case 0:
            hr = itemDef->put_ID(CComBSTR(L"DemoCplusplusArcmapAddin.DemoCommand"));
            break;
        case 1:
            hr = itemDef->put_ID(CComBSTR(L"DemoCplusplusArcmapAddin.DemoTool"));
            break;
        default:
            return E_INVALIDARG;
        }
        if (FAILED(hr)) return hr;
        hr = itemDef->put_Group(VARIANT_FALSE);
        if (FAILED(hr)) return hr;
        return E_NOTIMPL;
    }
    STDMETHOD(get_Name)(BSTR * Name)
    {
        *Name = ::SysAllocString(_T("C++DemoToolbar"));
        return S_OK;
    }
    STDMETHOD(get_Caption)(BSTR * Name)
    {
        *Name = ::SysAllocString(_T("C++DemoToolbar"));
        return S_OK;
    }
};

停靠窗口显隐开关

添加一个显示和隐藏窗体的命令，这里直接修改DemoCommand，将之作为可停靠窗体的显隐开关。

    STDMETHOD(OnClick)()
    {
        USING_MFC;
        //INFORMATION_MESSAGE_BOX(_T("C++Demo命令"));
        IDispatchPtr ipDispath;
        m_ipHookHelper->get_Hook(&ipDispath);
        IDockableWindowManagerPtr ipDockableWindowManager = ipDispath;

        IUIDPtr ipUID(CLSID_UID);

        TCHAR* pszClsid;
        ::StringFromCLSID(CLSID_DemoDockableWindow, &pszClsid);
        CComVariant vVariant(pszClsid);
        ipUID->put_Value(vVariant);

        //ipUID->put_Value(_variant_t(L"{D9A9C2BB-854C-4F0F-83F0-DD5B8FA3FE81}"));

        IDockableWindowPtr ipDockableWindow;
        HRESULT hr = ipDockableWindowManager->GetDockableWindow(ipUID, &ipDockableWindow);
        if (hr == S_OK&&ipDockableWindow != NULL)
        {
            VARIANT_BOOL vbVisible = VARIANT_FALSE;
            ipDockableWindow->IsVisible(&vbVisible);
            if (vbVisible == VARIANT_FALSE)
                ipDockableWindow->Show(VARIANT_TRUE);
            else
                ipDockableWindow->Show(VARIANT_FALSE);
        }
        return S_OK;
    }

// DemoDockableWindow.h : CDemoDockableWindow 的声明

#pragma once
#include "resource.h"       // 主符号



#include "DemoCplusplusArcmapAddin_i.h"
#include "DlgDockableWindow.h"



#if defined(_WIN32_WCE) && !defined(_CE_DCOM) && !defined(_CE_ALLOW_SINGLE_THREADED_OBJECTS_IN_MTA)
#error "Windows CE 平台(如不提供完全 DCOM 支持的 Windows Mobile 平台)上无法正确支持单线程 COM 对象。定义 _CE_ALLOW_SINGLE_THREADED_OBJECTS_IN_MTA 可强制 ATL 支持创建单线程 COM 对象实现并允许使用其单线程 COM 对象实现。rgs 文件中的线程模型已被设置为“Free”，原因是该模型是非 DCOM Windows CE 平台支持的唯一线程模型。"
#endif

using namespace ATL;


// CDemoDockableWindow

class ATL_NO_VTABLE CDemoDockableWindow :
    public CComObjectRootEx,
    public CComCoClass,
    public IDemoDockableWindow,
    public IDockableWindowDef
{
public:
    CDemoDockableWindow() :m_pDlg(NULL)
    {
    }

    DECLARE_REGISTRY_RESOURCEID(IDR_DEMODOCKABLEWINDOW)


    BEGIN_COM_MAP(CDemoDockableWindow)
        COM_INTERFACE_ENTRY(IDemoDockableWindow)
        COM_INTERFACE_ENTRY(IDockableWindowDef)
    END_COM_MAP()

    BEGIN_CATEGORY_MAP(CDemoDockableWindow)
        IMPLEMENTED_CATEGORY(__uuidof(CATID_MxDockableWindows))
        IMPLEMENTED_CATEGORY(__uuidof(CATID_GxDockableWindows))
        IMPLEMENTED_CATEGORY(__uuidof(CATID_GMxDockableWindows))
        IMPLEMENTED_CATEGORY(__uuidof(CATID_SxDockableWindows))
    END_CATEGORY_MAP()

    DECLARE_PROTECT_FINAL_CONSTRUCT()

    HRESULT FinalConstruct()
    {
        return S_OK;
    }

    void FinalRelease()
    {
    }
private:
    CDlgDockableWindow* m_pDlg;
    IHookHelperPtr m_ipHookHelper;
public:




    // IDockableWindowDef Methods
public:
    STDMETHOD(OnCreate)(LPDISPATCH hook)
    {
        m_ipHookHelper.CreateInstance(CLSID_HookHelper);
        HRESULT hr = m_ipHookHelper->putref_Hook(hook);
        return S_OK;
    }
    STDMETHOD(get_ChildHWND)(OLE_HANDLE * hWnd)
    {
        USING_MFC;
        if (m_pDlg == NULL)
        {
            m_pDlg = new CDlgDockableWindow();
            m_pDlg->Create(CDlgDockableWindow::IDD);
        }
        *hWnd = (OLE_HANDLE)m_pDlg->m_hWnd;
        return S_OK;
    }
    STDMETHOD(get_Name)(BSTR * Name)
    {
        *Name = ::SysAllocString(L"DemoDockableWindow");
        return S_OK;
    }
    STDMETHOD(get_Caption)(BSTR * Caption)
    {
        *Caption = ::SysAllocString(L"Demo停靠窗体");
        return S_OK;
    }
    STDMETHOD(OnDestroy)()
    {
        USING_MFC;
        if (m_pDlg != NULL&&TRUE == ::IsWindow(m_pDlg->GetSafeHwnd()))
        {
            m_pDlg->DestroyWindow();
            m_pDlg = NULL;
        }
        return S_OK;
    }
    STDMETHOD(get_UserData)(VARIANT * data)
    {
        ::VariantInit(data);
        return S_OK;
    }
};

OBJECT_ENTRY_AUTO(__uuidof(DemoDockableWindow), CDemoDockableWindow)
