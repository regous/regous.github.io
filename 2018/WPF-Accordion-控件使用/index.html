新建一个新的，WPF应用程序，我将之命名为 DemoAccordion

添加WPFToolkit的引用，可以在NuGet中联机里输入DotNetProjects，往下拉即可以找到，也可以去Github上下载源码自己编译，但这样不太好。

大致的效果已经出来了，但是依然还有几个明显不完美的地方：

  • 背景蓝色亮瞎
  • ScrollViewer没有生效，控件超出了界面
  • 子控件头的文字没有居中，相对箭头明显下沉

下面我们来分别解决这三个问题。

解决展开超出界面的问题

目前我能找到的解决方法是，在Accordion尺寸变化时，限制AccordionItem的最大高度。

<toolkit:Accordion x:Name="wndAccordion" VerticalContentAlignment="Stretch" SelectionMode="ZeroOrOne" SizeChanged="wndAccordion_SizeChanged" Width="{Binding Path=ActualWidth, RelativeSource={RelativeSource AncestorType=Grid}}" Height="{Binding Path=ActualHeight, RelativeSource={RelativeSource AncestorType=Grid}}" VerticalAlignment="Stretch">
private void wndAccordion_SizeChanged(object sender, SizeChangedEventArgs e)
{
    Accordion aAccordion = sender as Accordion;
    if (aAccordion == null || e.HeightChanged == false)
        return;
    SetAccordionItemMaxHeight(e.NewSize.Height);
}

private void SetAccordionItemMaxHeight(double dAccordionHeight)
{
    Accordion aAccordion = wndAccordion;
    if (aAccordion.Items.Count == 0)
        return;
    double dHeight = dAccordionHeight - 2;
    double dMaxHeight = 0;
    if (aAccordion.Items.Count == 1)
        dMaxHeight = dHeight;
    else
    {
        double dUnitHeight = 0;
        for (int i = 0; i < aAccordion.Items.Count; i++)
        {
            AccordionItem aItem = aAccordion.Items[i] as AccordionItem;
            if (aItem.IsSelected == false)
            {
                dUnitHeight = aItem.ActualHeight;
                break;
            }
        }
        dMaxHeight = dHeight - dUnitHeight * (aAccordion.Items.Count - 1);
    }
    for (int i = 0; i < aAccordion.Items.Count; i++)
    {
        AccordionItem aItem = aAccordion.Items[i] as AccordionItem;
        aItem.MaxHeight = dMaxHeight;
    }
}
